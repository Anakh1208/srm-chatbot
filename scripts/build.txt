def save_chunks_to_supabase(texts: list) -> bool:
    """Save chunks to Supabase database"""
    try:
        SUPABASE_URL = os.getenv("SUPABASE_URL")
        SUPABASE_KEY = os.getenv("SUPABASE_KEY")
        
        if not SUPABASE_URL or not SUPABASE_KEY:
            print("⚠️ Supabase credentials missing - saving locally only")
            return False
        
        supabase = create_client(SUPABASE_URL, SUPABASE_KEY)
        
        # Prepare data for batch insert
        documents = [
            {
                "content": text,
                "source_url": f"chunk_{i}"
            }
            for i, text in enumerate(texts)
        ]
        
        # Clear existing data (optional - remove if you want to append)
        supabase.table("documents").delete().neq("id", "00000000-0000-0000-0000-000000000000").execute()
        
        # Batch insert (Supabase supports up to 1000 rows per insert)
        batch_size = 1000
        for i in range(0, len(documents), batch_size):
            batch = documents[i:i + batch_size]
            response = supabase.table("documents").insert(batch).execute()
            print(f"   Uploaded {min(i + batch_size, len(documents))}/{len(documents)} chunks to Supabase")
        
        print(f"✅ Saved {len(texts)} chunks to Supabase")
        return True
        
    except Exception as e:
        print(f"❌ Supabase save failed: {str(e)}")
        return False


def load_chunks_from_supabase() -> list:
    """Load chunks from Supabase database"""
    try:
        SUPABASE_URL = os.getenv("SUPABASE_URL")
        SUPABASE_KEY = os.getenv("SUPABASE_KEY")
        
        if not SUPABASE_URL or not SUPABASE_KEY:
            print("⚠️ Supabase credentials missing - using local storage")
            return None
        
        supabase = create_client(SUPABASE_URL, SUPABASE_KEY)
        
        # Fetch all documents
        response = supabase.table("documents").select("*").execute()
        
        if response.data:
            print(f"✅ Loaded {len(response.data)} chunks from Supabase")
            
            # Convert to standard format
            chunks = [
                {
                    'text': doc['content'],
                    'metadata': {
                        'chunk_id': doc['id'],
                        'source': doc.get('source_url', 'unknown')
                    }
                }
                for doc in response.data
            ]
            return chunks
        else:
            print("⚠️ No documents found in Supabase")
            return None
            
    except Exception as e:
        print(f"❌ Supabase load failed: {str(e)}")
        return None


def load_chunks(chunks_path: str) -> list:
    """Load chunks from JSON"""
    print(f"\n📂 Loading chunks from: {chunks_path}")
    
    with open(chunks_path, 'r', encoding='utf-8') as f:
        data = json.load(f)
    
    if isinstance(data, dict) and 'chunks' in data:
        chunks = data['chunks']
    elif isinstance(data, list):
        chunks = data
    else:
        chunks = list(data.values()) if isinstance(data, dict) else data
    
    print(f"✅ Loaded {len(chunks)} chunks")
    return chunks


def normalize_chunks(chunks: list) -> list:
    """Normalize chunk format"""
    print(f"\n🔄 Normalizing chunks...")
    normalized = []
    
    for i, chunk in enumerate(chunks):
        if isinstance(chunk, dict):
            text = chunk.get('text') or chunk.get('chunk_text') or chunk.get('content', '')
            if text:
                normalized.append({
                    'text': text,
                    'metadata': chunk.get('metadata', {'chunk_id': i})
                })
        elif isinstance(chunk, str):
            normalized.append({'text': chunk, 'metadata': {'chunk_id': i}})
    
    print(f"✅ Normalized {len(normalized)} chunks")
    return normalized


def main():
    print("\n" + "="*70)

   # Paths

# Priority: Try merged chunks first (DB + web data)

if not chunks_path.exists():

# Fallback 2: Original chunks
if not chunks_path.exists():
    print("⚠️ Improved chunks not found, trying original chunks...")
    chunks_path = project_root / "data" / "processed" / "chunks.json"

vectorstore_dir = str(project_root / "data" / "vectorstore")

try:
    # Step 1: Load chunks (try Supabase first, fallback to local)

    chunks = load_chunks_from_supabase()
    
    if not chunks:

        print("📂 Falling back to local storage...")

    if not normalized_chunks:
        print("❌ No chunks to process!")
        sys.exit(1)

    print("\n🔧 Initializing embeddings generator...")
    embeddings_generator = EmbeddingsGenerator()
    
    # Generate embeddings
    print(f"\n🔢 Generating embeddings...")

    all_embeddings = []
    batch_size = 32
    for i in range(0, len(texts), batch_size):
        batch = texts[i:i + batch_size]
        batch_embs = embeddings_generator.embed(batch)

    embeddings = np.array(all_embeddings, dtype=np.float32)

    vector_store = FAISSVectorStore()

    os.makedirs(vectorstore_dir, exist_ok=True)
    vector_store.save_index(vectorstore_dir)
    
    # Save chunks locally (backup)

    with open(chunks_file, 'w', encoding='utf-8') as f:
